 49def add_node(G: nx.MultiDiGraph, node: "Node", config: dict) -> None:
 50	"""Add a node to the graph with the given type and optional URL."""
 51	# if node is not present, add it
 52	if node not in G:
 53		# add the node
 54		G.add_node(
 55			node,  # `Node` object, `str(node)` will be the key
 56			rank=node.get_rank(),  # for ranking/ordering of the nodes
 57			**config["node"][
 58				node.node_type
 59			],  # attributes (color, shape, etc) for the node type
 60		)
 61		# add a URL -- doesn't work for images
 62		if node.url:
 63			G.nodes[node]["URL"] = f'"{node.url}"'
 64	else:
 65		# if it's already present, we have a key duplication
 66		raise ValueError(f"node {node.path} already exists in the graph!")

 69def classify_node(path: str, root: str = ".") -> NodeType:
 70	# posixify path
 71	path = normalize_path(path)
 72	# path relative to the root module
 73	rel_path: str = os.path.relpath(path, root)
 74	# parent directory (module it's directly in)
 75	parent_dir: str = os.path.dirname(rel_path)
 76	# if it's in the root, set the parent dir to "."
 77	if parent_dir == "":
 78		parent_dir = "."
 79
 80	# error checking
 81	if not os.path.exists(path):
 82		raise FileNotFoundError(f"file not found: '{path}' for {path = } and {root = }")
 83
 84	if not os.path.exists(parent_dir):
 85		raise FileNotFoundError(
 86			f"parent directory not found: '{parent_dir}' for {path = } and {root = }"
 87		)
 88
 89	if not os.path.exists(rel_path):
 90		raise FileNotFoundError(
 91			f"relative path not found: '{rel_path}' for {path = } and {root = }"
 92		)
 93
 94	# if directory
 95	if os.path.isdir(path):
 96		files: list[str] = os.listdir(path)
 97		# handle root
 98		if rel_path == ".":
 99			if "__init__.py" in files:
100				return "module_root"
101			else:
102				return "root"
103		else:
104			# module or ordinary directory
105			return "module_dir" if "__init__.py" in files else "dir"
106	elif rel_path.endswith(".py"):
107		# if a py file, module or script
108		if "__init__.py" in os.listdir(parent_dir):
109			return "module_file"
110		else:
111			return "script"
112	else:
113		raise ValueError(f"unknown path type: {path}")

127	@classmethod
128	def get_node(
129		cls,
130		path: str,
131		config: dict,
132		root: str = ".",
133	) -> "Node":
134		if path == "":
135			path = "."
136
137		# set up aliases
138		aliases: set[str] = {path}
139
140		# path relative to root
141		rel_path: str = normalize_path(os.path.relpath(path, root))
142		aliases.add(rel_path)
143		if os.path.basename(rel_path) == "__init__.py":
144			rel_path = os.path.dirname(rel_path).removesuffix("/")
145			aliases.add(rel_path)
146
147		# node type for formatting
148		node_type: NodeType = classify_node(path, root)
149
150		# get parent dir
151		parent_dir: str = normalize_path(os.path.dirname(rel_path))
152		if not parent_dir:
153			parent_dir = "."
154
155		# unique display name
156		display_name: str
157		if node_type.startswith("module"):
158			display_name = path_to_module(rel_path, strict_names=config["graph"]["strict_names"])
159			# special case for when there is a file/module with the same name as the package
160			if node_type != "module_root":
161				display_name = augment_module_name(display_name, config)
162		else:
163			display_name = rel_path
164
165		aliases.add(display_name)
166		if node_type in {"module_root", "root"}:
167			display_name = config["root_node_name"]
168			parent_dir = None
169			aliases.add(display_name)
170
171		# get url if needed
172		url: str | None = None
173		url_prefix: str = config["url_prefix"]
174		if url_prefix is not None:
175			url = f"{url_prefix}{rel_path}"
176			if config.get("auto_url_replace", None):
177				for k, v in config["auto_url_replace"].items():
178					url = url.replace(k, v)
179
180		# assemble and return node
181		node: Node = Node(
182			orig_path=path,
183			rel_path=rel_path,
184			aliases=aliases,
185			display_name=display_name,
186			config=config,
187			url=url,
188			node_type=node_type,
189			parent_dir=parent_dir,
190		)
191
192		return node

239def build_graph(
240	root: str,
241	config: dict,
242) -> nx.MultiDiGraph:
243	# process config
244	# --------------------------------------------------
245	include_local_imports: bool = config["graph"]["include_local_imports"]
246	edge_config: dict[str, Any] = config["edge"]
247
248	# create graph, get dirs and package name
249	# --------------------------------------------------
250	G: nx.MultiDiGraph = nx.MultiDiGraph()
251	directories: set[str] = get_relevant_directories(root)
252	package_name: str = os.path.basename(os.path.abspath(root))
253	assert package_name == config["PACKAGE_NAME"], f"{package_name = }, {config['PACKAGE_NAME'] = }"
254
255	# Add nodes for directories and root
256	# --------------------------------------------------
257	directory_nodes: dict[str, Node] = {
258		augment_module_name(directory, config) : Node.get_node(directory, config=config)
259		for directory in directories
260	}
261	for node in directory_nodes.values():
262		add_node(G, node, config=config)
263
264	# add folder hierarchy edges
265	# --------------------------------------------------
266	for directory, node in directory_nodes.items():
267		# no parent of the root
268		if node.is_root():
269			continue
270
271		# get parent node
272		parent_node: Node = directory_nodes[node.parent_dir]
273
274		# figure out edge type -- different styles for different categories
275		edge_type: str
276		if parent_node.is_module() and node.is_module():
277			edge_type = "module_hierarchy"
278		else:
279			edge_type = "hierarchy"
280
281		# add edge to graph
282		G.add_edge(parent_node, node, **edge_config[edge_type])
283
284	# get python files
285	# --------------------------------------------------
286	python_files: list[str] = get_python_files(root)
287
288	# add files nodes and heirarchy edges to folders
289	# --------------------------------------------------
290	nodes_dict: dict[str, Node] = dict()
291	for python_file in python_files:
292		node: Node
293		# special handling for init files
294		is_init: bool = python_file.endswith("__init__.py") or python_file == "."
295		if is_init:
296			augmented_module_name: str = augment_module_name(
297				os.path.dirname(python_file) or ".",
298				config,
299			)
300			node = directory_nodes[augmented_module_name]
301		else:
302			node = Node.get_node(python_file, config=config)
303			add_node(G, node, config=config)
304
305		# this will add the directory node if it doesn't exist
306		nodes_dict[node.display_name] = node
307
308		# add file hierarchy
309		if edge_config["hierarchy"]:
310			if node.node_type not in {"root", "module_root"} and not is_init:
311				parents: list[str] = [
312					parent_node.display_name
313					for parent_node in directory_nodes.values()
314					if node.parent_dir == parent_node.rel_path
315				]
316				if parents:
317					assert (
318						len(parents) == 1
319					), f"multiple parents found for {node.orig_path = }: {parents}"
320					module_parent: str = sorted(parents, key=len)[-1]
321					if module_parent == config["root_node_name"]:
322						module_parent = "."
323
324					parent_node: Node
325					try:
326						parent_node = directory_nodes[module_parent]
327					except KeyError as e:
328						if config["graph"]["except_if_missing_edges"]:
329							raise KeyError(
330								f"missing parent node for {node.orig_path = }: '{module_parent}'. if you think this is a mistake, set `graph.except_if_missing_edges` to `False` in the config"
331							) from e
332						else:
333							warnings.warn(
334								f"missing parent node for {node.orig_path = }: '{module_parent}'"
335							)
336
337					G.add_edge(
338						parent_node,
339						node,
340						**edge_config["hierarchy"],
341					)
342
343	# add import edges
344	# --------------------------------------------------
345	if include_local_imports:
346		print("!!!!!!!!!! INCLUDING LOCAL IMPORTS")
347		# init empty lists, cant modify while iterating
348		# -------------------------
349		nodes_to_add: list[dict] = []
350		edges_to_add: list[dict] = []
351		for node_key in G.nodes:
352
353			# get and check the node
354			# -------------------------
355			node: Node
356			if isinstance(node_key, Node):
357				node = node_key
358			else:
359				raise ValueError(
360					f"unknown node type: {node_key = }, {type(node_key) = }"
361				)
362
363			# Read source code
364			# # -------------------------
365			node_path: str = node.orig_path
366			if os.path.isdir(node_path):
367				node_path = os.path.join(node_path, "__init__.py")
368			elif node_path == ".":
369				node_path = "__init__.py"
370
371			try:
372				with open(node_path, "r", encoding="utf-8") as f:
373					source_code: str = f.read()
374			except FileNotFoundError as e:
375				if config["graph"]["except_if_missing_edges"]:
376					raise FileNotFoundError(
377						f"could not read source code for {node_path = }. if you think this is a mistake, set `graph.except_if_missing_edges` to `False` in the config"
378					) from e
379				else:
380					warnings.warn(f"could not read source code for {node_path = }, skipping")
381					continue
382
383			# Get imports, dedupe, and loop over them
384			# -------------------------
385			imported_modules: list[str] = list(set(
386				get_imports(source_code, allow_missing_imports=not config["graph"]["except_if_missing_edges"])
387			))
388
389			for imported_module in imported_modules:
390				# Convert import to module name
391				imported_module_name = imported_module
392
393
394
395				# if stripping module prefix, remove it
396				if config["graph"]["strip_module_prefix"]:
397					imported_module_name = imported_module_name.removeprefix(
398						package_name
399					).removeprefix(".")
400					if not imported_module_name:
401						# if empty string, it means we are looking for the root
402						imported_module_name = config["root_node_name"]
403
404				if imported_module_name in nodes_dict:
405		
406					# adding edge to local import
407					# -------------------------
408					edge_type = (
409						"inits"
410						if node_path.endswith("__init__.py")
411						else "uses"
412					)
413					if edge_config.get(edge_type):
414						edges_to_add.append(
415							dict(
416								u_for_edge=nodes_dict[imported_module_name],
417								v_for_edge=node,
418								**edge_config[edge_type],
419							)
420						)
421				else:
422					# assume external module
423					# -------------------------
424					if config["graph"]["include_externals"]:
425						nodes_to_add.append(
426							dict(
427								node_for_adding=imported_module_name,
428								rank=0,  # for ranking/ordering of the nodes
429								**config["node"][
430									"external"
431								],  # attributes (color, shape, etc) for the node type
432							)
433						)
434						edges_to_add.append(
435							dict(
436								u_for_edge=imported_module_name,
437								v_for_edge=node,
438								**edge_config["external"],
439							)
440						)
441
442		# add the nodes and edges we were missing
443		# -------------------------
444
445		# these nodes should only be present if we are including external imports of other packages
446		for x in nodes_to_add:
447			G.add_node(**x)
448
449		for x in edges_to_add:
450			G.add_edge(**x)
451
452	return G

462def main(
463	root: str | None = None,
464	module: str | None = None,
465	output: str = "output",
466	output_fmt: Literal["svg", "png"] = "svg",
467	config_file: str | None = None,
468	print_cfg: bool = False,
469	verbose: bool = False,
470	**kwargs,
471) -> None:
472	"""Main function to generate and render a graphviz DOT file representing module dependencies
473
474	# Keyword-only arguments
475	- `root: str` (this arg or `module` is REQUIRED)
476	    root directory to search for Python files
477	- `module: str` (this arg or `root` is REQUIRED)
478		name of module to generate graph for -- must be importable in the current environment
479	- `output: str`
480	    output filename (without extension)
481	    default: `"output"`
482	- `output_fmt: Literal["svg", "png"]`
483	    output format for running `dot`
484	    default: `"svg"`
485	- `config_file: str | None = None`
486	    path to a JSON file containing configuration options
487	- `print_cfg: bool = False`
488	    whether to print the configuration after loading it -- if this is set, the program will exit after printing the config
489	- `verbose: bool = False`
490	- `h` or `help`
491	    print this help message and exit
492
493	# Configuration options
494	either specify these in a json file, or separate levels with a dot. set to `None` to disable.
495	- `url_prefix: str|None`
496	    manually add a prefix to the url. if set to `None`, will try to auto-detect from git
497	- `auto_url_format: str|None`
498	    how to format a url given the git remote url and branch name.
499	    default: "{git_remote_url}/tree/{git_branch}/" (which works for github)
500	    set to `None` to disable. if both this and `url_prefix` are none, the svg will not have URLs
501	- `auto_url_replace: dict[str,str]`
502	    string-replace pairs to apply to the git remote url before formatting it
503	    note: trailing slashes are also stripped
504	    default: `{".git": ""}`
505	- `node.dir: dict|None`
506	    kwargs for directory nodes
507	- `node.file: dict|None`
508	    kwargs for file nodes
509	- `edge.hierarchy: dict|None`
510	    kwargs for hierarchy edges (i.e. file A is part of module B)
511	- `edge.uses: dict|None`
512	    kwargs for uses edges (i.e. file A imports module B for using it)
513	- `edge.inits: dict|None`
514	    kwargs for init edges (i.e. __init__.py file imports something from downstream of itself)
515	- `dot_attrs: dict`
516	    kwargs for the dot graph itself
517	    default: `{'rankdir': 'TB'}` (top to bottom)
518	    you can change it via `dot_attrs.rankdir=LR` for left to right, etc.
519
520	To modify an element of a dict without specifying the whole dict, you can use "." as a level separator:
521	```
522	--edge.uses.color=green
523	```
524
525	"""
526
527	# handle kwargs and config
528	# --------------------------------------------------
529
530	# handle module vs explicit path
531	if root is None:
532		assert module is not None, f"either root or module must be given, got values for both: {root = }, {module = }"
533		root = get_module_directory(module)
534	elif module is None:
535		assert root is not None, f"either root or module must be given, got values for both: {root = }, {module = }"
536	else:
537		raise ValueError("either root or module must be given, got `None` for both")
538	
539
540	# update config from file if given
541
542	CONFIG: dict = deepcopy(_DEFAULT_CONFIG)
543	print(kwargs)
544	print(CONFIG["graph"])
545	if config_file is not None:
546		with open(config_file, "r", encoding="utf-8") as f:
547			update_with_nested_dict(CONFIG, json.load(f))
548
549	# update config from kwargs
550	if len(kwargs) > 0:
551		update_with_nested_dict(
552			CONFIG,
553			kwargs_to_nested_dict(
554				kwargs, transform_key=lambda x: x.lstrip("-"), sep="."
555			),
556		)
557	print(CONFIG["graph"])
558
559	# process by converting none types, auto-detecting url_prefix from git if needed
560	# special config processing: if we are doing a module, then we try to get the url prefix from there
561	url_prefix: str | None = None
562	if module is not None and CONFIG["url_prefix"] is None:
563		url_prefix = get_package_repository_url(module)
564	_process_config(CONFIG, root=root)
565	if url_prefix is not None:
566		CONFIG["url_prefix"] = url_prefix
567
568	# print help message and exit
569	if "h" in CONFIG or "help" in CONFIG:
570		print(main.__doc__)
571		print("# current config:")
572		print(json.dumps(CONFIG, indent=2))
573		exit()
574
575	# print config and exit
576	if print_cfg:
577		print(json.dumps(CONFIG, indent=2))
578		exit()
579
580	if root is None:
581		raise ValueError("root is required")
582	
583	# set up some other globals
584	# --------------------------------------------------
585
586	# get global package name, set root node name if neededME
587	CONFIG["PACKAGE_NAME"] = os.path.basename(os.path.abspath(root))
588	if not CONFIG["graph"]["strip_module_prefix"]:
589		CONFIG["root_node_name"] = CONFIG["PACKAGE_NAME"]
590	
591	# move directory, build graph, move back
592	# --------------------------------------------------
593
594	# change directory
595	orig_dir: str = os.getcwd()
596	os.chdir(root)
597
598	print("# building graph...")
599	G: nx.MultiDiGraph = build_graph(
600		# pass "." since we just moved to the root directory
601		root=".",
602		config = CONFIG,
603	)
604	print(f"\t built graph with {len(G.nodes)} nodes and {len(G.edges)} edges")
605
606	# change back to original directory
607	os.chdir(orig_dir)
608
609
610	# output
611	# --------------------------------------------------
612

614	output_file_dot: str = f"{output}.dot"
615	print(f"# writing dot file: {output_file_dot}")
616	write_dot(G, f"{output_file_dot}", dot_attrs=CONFIG["dot_attrs"])
617
618	if output_fmt == "html":
619		# if HTML, generate it and put the dotfile contents inline
620		print("# generating html...")
621		from dep_graph_viz.html import generate_html
622
623		generate_html(output_file_dot, f"{output}.html")
624	else:
625		# otherwise, run dot/graphviz and convert to desired format
626		print("# running dot...")
627		print("\tcommand:")
628		cmd: str = f"dot -T{output_fmt} {output_file_dot} -o {output}.{output_fmt} {'-v' if verbose else ''}"
629		print(f"\t$ {cmd}")
630		subprocess.run(cmd, check=True)
631
632	print("# done!")

Main function to generate and render a graphviz DOT file representing module dependencies

Keyword-only arguments

  • root: str (this arg or module is REQUIRED) root directory to search for Python files
  • module: str (this arg or root is REQUIRED) name of module to generate graph for -- must be importable in the current environment
  • output: str output filename (without extension) default: "output"
  • output_fmt: Literal["svg", "png"] output format for running dot default: "svg"
  • config_file: str | None = None path to a JSON file containing configuration options
  • print_cfg: bool = False whether to print the configuration after loading it -- if this is set, the program will exit after printing the config
  • verbose: bool = False
  • h or help print this help message and exit

Configuration options

either specify these in a json file, or separate levels with a dot. set to None to disable.

  • url_prefix: str|None manually add a prefix to the url. if set to None, will try to auto-detect from git
  • auto_url_format: str|None how to format a url given the git remote url and branch name. default: "{git_remote_url}/tree/{git_branch}/" (which works for github) set to None to disable. if both this and url_prefix are none, the svg will not have URLs
  • auto_url_replace: dict[str,str] string-replace pairs to apply to the git remote url before formatting it note: trailing slashes are also stripped default: {".git": ""}
  • node.dir: dict|None kwargs for directory nodes
  • node.file: dict|None kwargs for file nodes
  • edge.hierarchy: dict|None kwargs for hierarchy edges (i.e. file A is part of module B)
  • edge.uses: dict|None kwargs for uses edges (i.e. file A imports module B for using it)
  • edge.inits: dict|None kwargs for init edges (i.e. __init__.py file imports something from downstream of itself)
  • dot_attrs: dict kwargs for the dot graph itself default: {'rankdir': 'TB'} (top to bottom) you can change it via dot_attrs.rankdir=LR for left to right, etc.

To modify an element of a dict without specifying the whole dict, you can use "." as a level separator:

--edge.uses.color=green
