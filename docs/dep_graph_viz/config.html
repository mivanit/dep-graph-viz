 90def _process_config(config: dict, root: str | None = ".") -> None:
 91	"""converts none types, auto-detects url_prefix from git if needed
 92
 93	- mapping null values: in CONFIG, a value under the `CONFIG["edge"]` or `CONFIG["node"]` dicts that matches `NULL_STRINGS` will be converted to `None`
 94	- auto-generating url: if `CONFIG["url_prefix"]` is `None`, `CONFIG["auto_url_format"]` is not `None`, and `root` is not `None`, the git remote url and branch will be auto-detected and formatted into a URL
 95
 96	# Parameters:
 97	 - `root : str`
 98	    place to look for git info. if `None`, will not try to auto-detect git info
 99	   (defaults to `"."`)
100
101	# Returns:
102	 - `None`
103
104	# Modifies:
105	global variable `CONFIG`, specifically:
106	 - `CONFIG["edge"][*]` and `CONFIG["node"][*]` which match `NULL_STRINGS` will be converted to `None`
107	 - `CONFIG["url_prefix"]` will be set to a formatted URL if it is `None` and `CONFIG["auto_url_format"]` is not `None`
108	"""
109
110	# convert none/null items
111	for k_conv in ("edge", "node"):
112		for key, value in config[k_conv].items():
113			if isinstance(value, str):
114				if value.lower() in NULL_STRINGS:
115					config[k_conv][key] = None
116			elif isinstance(value, dict):
117				for sub_key, sub_value in value.items():
118					if isinstance(sub_value, str) and sub_value.lower() in NULL_STRINGS:
119						config[k_conv][key][sub_key] = None
120						print(f"converted {k_conv}.{key}.{sub_key} to None")
121
122	# get git url and branch
123	if (
124		(config["url_prefix"] is None)
125		and (config["auto_url_format"] is not None)
126		and (root is not None)
127	):
128		try:
129			# navigate to root
130			orig_dir: str = os.getcwd()
131			os.chdir(root)
132			# get git remote url
133			git_remote_url: str = (
134				subprocess.check_output(
135					"git remote get-url origin",
136					shell=True,
137					encoding="utf-8",
138				)
139				.strip()
140				.rstrip("/")
141			)
142			for rep_key, rep_val in config["auto_url_replace"].items():
143				git_remote_url = git_remote_url.replace(rep_key, rep_val)
144			# get branch
145			git_branch: str = subprocess.check_output(
146				"git rev-parse --abbrev-ref HEAD",
147				shell=True,
148				encoding="utf-8",
149			).strip()
150			config["url_prefix"] = config["auto_url_format"].format(
151				git_remote_url=git_remote_url, git_branch=git_branch
152			)
153		except subprocess.CalledProcessError as e:
154			print(f"could not get git info, not adding URLs: {e}")
155			config["url_prefix"] = None
156		finally:
157			# go back to original directory
158			os.chdir(orig_dir)

NULL_STRINGS: set[str] = {'null', 'none'}
