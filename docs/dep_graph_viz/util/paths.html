 16def path_to_module(path: str, strict_names: bool = True) -> str:
 17	"convert a path to a python file to a module name"
 18	# First normalize the path
 19	norm_path: str = normalize_path(path).removesuffix(".py").removeprefix("/")
 20
 21	if not norm_path:
 22		raise ValueError(f"empty path: '{path}'")
 23
 24	# Check for dots in path components (not allowed)
 25	if "." in norm_path:
 26		# root path is a special case
 27		if norm_path == ".":
 28			return norm_path
 29		else:
 30			raise ValueError(f"path contains '.', invalid: '{path}'")
 31
 32	try:
 33		# Validate each path component
 34		components = norm_path.split("/")
 35		for component in components:
 36			# Check empty components
 37			if not component:
 38				raise ValueError(f"empty path component in: '{path}'")
 39
 40			# Check for invalid characters in module names
 41			if not re.match(MODULE_NAME_REGEX, component):
 42				raise ValueError(
 43					f"invalid module name component: '{component}' in path: '{path}'"
 44				)
 45
 46			# Check for control characters
 47			if any(ord(c) < 32 or ord(c) == 127 for c in component):
 48				raise ValueError(f"control character in module name: '{path}'")
 49
 50			# Check length
 51			if len(component) > 255:
 52				raise ValueError(f"module name component too long: '{component}'")
 53
 54			# Windows reserved names (case-insensitive)
 55			reserved_names = {
 56				"con",
 57				"prn",
 58				"aux",
 59				"nul",
 60				"com1",
 61				"com2",
 62				"com3",
 63				"com4",
 64				"com5",
 65				"com6",
 66				"com7",
 67				"com8",
 68				"com9",
 69				"lpt1",
 70				"lpt2",
 71				"lpt3",
 72				"lpt4",
 73				"lpt5",
 74				"lpt6",
 75				"lpt7",
 76				"lpt8",
 77				"lpt9",
 78			}
 79			if component.lower() in reserved_names:
 80				raise ValueError(f"invalid Windows reserved name: '{component}'")
 81
 82			# Special names like .hidden files
 83			if component.startswith("."):
 84				raise ValueError(f"module name cannot start with dot: '{component}'")
 85
 86			# Numeric start check
 87			if component[0].isdigit():
 88				raise ValueError(f"module name cannot start with number: '{component}'")
 89	except ValueError as e:
 90		if strict_names:
 91			raise ValueError(f"invalid path: '{path}'. If you think this is a mistake, set `graph.strict_names` to `False` in the config. see error:\n\t{e}") from e
 92		else:
 93			warnings.warn(f"invalid path: '{path}', continuing because `graph.strict_names` is set to `False` in the config. see error:\n\t{e}")
 94
 95	return norm_path.replace("/", ".")

 99def get_module_directory(module_name: str) -> str:
100    """Get the directory containing a module's source code.
101    
102    Args:
103        module_name: Name of module as you would use in an import statement
104        
105    Returns:
106        Absolute path to the directory containing the module
107        
108    Raises:
109        ImportError: If module cannot be imported
110        AttributeError: If module does not have a __file__ attribute
111    """
112    module = importlib.import_module(module_name)
113    
114    # Get the module's file path
115    if not hasattr(module, '__file__'):
116        raise AttributeError(f"Module {module_name} has no __file__ attribute")
117        
118    module_file = module.__file__
119    
120    # Get directory containing the module
121    module_dir = os.path.dirname(os.path.abspath(module_file))
122    
123    return module_dir

126def get_package_repository_url(package_name: str) -> str|None:
127    """Get the repository URL for a Python package.
128    
129    Tries multiple methods:
130    1. package metadata "project_urls" under Repository/Source/Code keys
131    2. package metadata "home_page" 
132    3. package metadata "download_url"
133    
134    Args:
135        package_name: Name of the installed package
136        
137    Returns:
138        Repository URL if found, None otherwise
139    
140    Raises:
141        importlib.metadata.PackageNotFoundError: If package is not installed
142    """
143    metadata = importlib.metadata.metadata(package_name)
144    
145    # Check project_urls first
146    if "project_urls" in metadata:
147        try:
148            urls = json.loads(metadata["project_urls"])
149            repo_keys = ["Repository", "Source", "Code", "Source Code", "Homepage"]
150            for key in repo_keys:
151                if key in urls:
152                    return urls[key]
153        except:
154            pass
155    
156    # Try home_page
157    if "home-page" in metadata:
158        return metadata["home-page"]
159            
160    # Try download_url
161    if "download-url" in metadata:
162        return metadata["download-url"]
163    
164    return None
